public class HeapSort {
    static void adjust(int[] A, int i, int n){
        int tmp = A[i];
        int j = 2*i + 1;
        while(j<n){
            if( j<n-1 && A[j]<A[j+1] ) j++;
            if( tmp > A[j] ) break;
            A[i] = A[j];
            i = j;
            j = 2*j + 1;
        }
        A[i] = tmp;
    }
    static void build(int[] A, int n){
        for(int i = n/2; i>=0; i--){
            adjust(A, i, n);
        }
    }
    static int[] heapSort(int[] A, int n){
        build(A, n);
        for(int i = n-1; i>=0; i--){
            int tmp = A[0];
            A[0] = A[i];
            A[i] = tmp;
            adjust(A, 0, i);
        }
        return A;
    }
}
