16. 3Sum Closest

Given an array S of n integers, find three integers in S such that the sum is closest to a given number, target. Return the sum of the three integers. You may assume that each input would have exactly one solution.

For example, given array S = {-1 2 1 -4}, and target = 1.

The sum that is closest to the target is 2. (-1 + 2 + 1 = 2).

解答：

class Solution {
public:
    int threeSumClosest(vector<int>& nums, int target) {
        int n = nums.size();
        int minGap = 0x7fffffff;
        sort(nums.begin(), nums.end());
        int res = 0;
        for(int i = 0; i<n; i++){
            int cur = nums[i];
            int l = i+1;
            int r = n-1;
            while(l<r){
                int sum = cur + nums[l] + nums[r];
                if(abs(sum -target) < minGap) {
                    minGap = abs(sum - target);
                    res = sum;
                }
                if(sum < target) l++;
                else r--;
            }
        }
        return res;
    }
};

17. Letter Combinations of a Phone Number

Given a digit string, return all possible letter combinations that the number could represent.

Input: Digit string “23”

Output: [“ad”, “ae”, “af”, “bd”, “be”, “bf”, “cd”, “ce”, “cf”].

解答：

class Solution {
public:
    char array[10][4] = {{' ',' ',' ',' '},{' ',' ',' ',' '},{'a', 'b', 'c', ' '},{'d','e','f', ' '},{'g','h','i', ' '},{'j','k','l', ' '},{'m','n','o', ' '}, {'p', 'q', 'r', 's'}, {'t','u', 'v', ' '}, {'w', 'x', 'y', 'z'}
    };
    vector<string> v;
    vector<string> letterCombinations(string digits) {
        int n = digits.length();
        if(n == 0) return v;
        string res = "";
        dfs(0, digits, res);
        return v;
    }
    void dfs(int idx, string s, string res){
        if(idx>=s.length()) {
            v.push_back(res);
            return;
        }
        int index = s[idx]-'0';
        for(int i = 0; i<4; i++){
            if(array[index][i] != ' ') {
                res += array[index][i];
                dfs(idx+1, s, res);
                res = res.substr(0, res.length()-1);
            }
        }
    }
}  ;

18. 4Sum

Given an array S of n integers, are there elements a, b, c, and d in S such that a + b + c + d = target? Find all unique quadruplets in the array which gives the sum of target.

Note: The solution set must not contain duplicate quadruplets. 

For example, given array S = [1, 0, -1, 0, -2, 2], and target = 0.
A solution set is:
[
  [-1,  0, 0, 1],
  [-2, -1, 1, 2],
  [-2,  0, 0, 2]
]

解答：

class Solution {
public:
    vector<vector<int>> fourSum(vector<int>& nums, int target) {
        int n = nums.size();
        vector<vector<int>> res;
        if(n<4) return res;
        sort(nums.begin(), nums.end());
        for(int i = 0; i<n-3; i++){
            for(int j = i+1; j<n-2; j++){
                int l = j+1;
                int r = n-1;
                while(l<r){
                    int sum = nums[i] + nums[j] + nums[l] + nums[r];
                    if(sum == target){
                        vector<int> v = {nums[i], nums[j], nums[l], nums[r]};
                        res.push_back(v);
                        //++l;
                        //--r;
                        while(++l && nums[l] == nums[l-1]);
                        while(--r && nums[r] == nums[r+1]);
                    }else if(sum<target) l++;
                    else r--;
                }
                while(j<n-2&&nums[j]==nums[j+1])
                    j++;
            }
            while(i<n-2 && nums[i] == nums[i+1])
                i++;
        }
        return res;
    }
};

22. Generate Parentheses

Given n pairs of parentheses, write a function to generate all combinations of well-formed parentheses. 

For example, given n = 3, a solution set is: 
[
  "((()))",
  "(()())",
  "(())()",
  "()(())",
  "()()()"
]

解答：

一种很慢的方法：

class Solution {
public:
    vector<string> v;
    vector<string> generateParenthesis(int n) {
        string res = "(";
        robot(1, n, res);
        return v;
    }
    void robot(int idx, int n, string res){
        if(idx == 2*n-1){
            res += ")";
            if(isValid(res)) v.push_back(res);
            return;
        }
        res += "(";
        robot(idx+1, n, res);
        res = res.substr(0, res.length()-1);
        res += ")";
        robot(idx+1, n, res);
    }
    bool isValid(string str) {
        int n = str.length();
        if(n == 0 || n%2 == 1) return false;
        stack<char> s;
        int i = 1;
        int j = i;
        s.push(str[0]);
        while(str[i] != '\0'){
            if(canPush(str[i])) {s.push(str[i]); j++;}
            else{
                if(!s.empty() && canPop(s.top(), str[i])) {s.pop(); j++;}
            }
            i++;
        }
        if(s.empty() && j==n) return true;
        return false;
    }
    bool canPop(char top, char check){
        if(top == '(' && check == ')' || top == '{' && check == '}' || top == '[' && check == ']')
            return true;
        return false;
    }
    bool canPush(char check){
        if(check == '(' || check == '{' || check == '[') return true;
        return false;
    }
};

快的方法：（卡特兰数）

class Solution {
public:
    vector<string> v;
    vector<string> generateParenthesis(int n) {
        robot(n, n, "");
        return v;
    }
    void robot(int l, int r, string res){
        if(l == 0 && r == 0) {
            v.push_back(res);
            return;
        }
        if( l > 0) robot(l-1, r, res+"(");
        if( r>0 && r>l) robot(l, r-1, res+")");
    }
};
