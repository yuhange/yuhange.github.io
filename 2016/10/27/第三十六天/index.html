51. N-Queens 52. N-Queens 2

N皇后问题的两种问法

解答：

class Solution {
public:
    vector<vector<string>> res;
    int *x;
    int sum = 0;
    vector<vector<string>> solveNQueens(int n) {
        x = new int[n+1];
        for(int i= 0;i<=n;i++)
            x[i] = 0;
        dfs(1, n);
        return res;
    }
    void dfs(int idx, int n){
        if(idx>n){
            //此次递归成功，记录
            sum++;
            vector<string> vec;
            for(int i = 1; i<=n; i++){
                string s = "";
                for(int j = 1; j<=n; j++){
                    if(j == x[i]) s+='Q';
                    else s+='.';
                }
                vec.push_back(s);
            }
            res.push_back(vec);
            return;
        }
        for(int i = 1; i<=n; i++){
            x[idx] = i;
            if(legal(idx)) dfs(idx+1, n);
        }
    }
    bool legal(int k){
        for(int j = 1;j<k;j++)
        if(abs(x[k] - x[j]) == abs(k-j)||x[j] == x[k])
            return false;
        return true;
    }
};

53. Maximum Subarray

动态规划问题

解答：

class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int n = nums.size();
        if(nums.size() == 0) return 0;
        if(nums.size() == 1) return nums[0];
        int res = nums[0];
        int sum = nums[0];
        for(int i = 1; i<n; i++){
            if(sum < 0) sum = nums[i] ;
            else sum += nums[i];
            res = max(sum, res);
        }
        return res;
    }
};

54. Spiral Matrix

螺旋打印数组问题

解答：

class Solution {
public:
    vector<int> res;
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        int row = matrix.size();
        if(row == 0) return res;
        int colum = matrix[0].size();
        int start = 0;
        while(row>start*2 && colum>start*2){
            helper(matrix, start, colum-1-start, row-1-start);
            start++;
        }
        return res;
    }
    void helper(vector<vector<int>>& m, int start,int eX, int eY){
        // cout<<sX<<" "<<sY<<" "<<eX<<" "<<eY<<endl;
        for(int i = start; i<=eX; i++) res.push_back(m[start][i]);
        if(start < eY)
            for(int i = start+1; i<=eY; i++) res.push_back(m[i][eX]);
        if(start < eX && start < eY)
            for(int i = eX-1; i>=start; i--) res.push_back(m[eY][i]);
        if(start < eX && start < eY-1)
            for(int i = eY-1; i>=start+1; i--) res.push_back(m[i][start]);
    }
};
