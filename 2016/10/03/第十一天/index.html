61. Rotate List

Given a list, rotate the list to the right by k places, where k is non-negative.

For example:

Given 1->2->3->4->5->NULL and k = 2,

return 4->5->1->2->3->NULL.

实现题。

注意，K可能会大于链表的长度。先计算出链表的长度，然后用K取余。

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* rotateRight(ListNode* head, int k) {
        if(head == NULL) return NULL;
        ListNode* t = head;
        int length = 0;
        while(t){
            t = t->next;
            length++;
        }
        if(length<=k) k %= length;
        if(k == 0) return head;
        ListNode* p = head;
        ListNode* fast = head;
        ListNode* slow = head;
        ListNode* pre = slow;
        while(fast->next && k){
            fast = fast->next;
            k--;
        }
        while(fast->next && slow->next){
            fast = fast->next;
            slow = slow->next;
            pre = slow;
        }
        slow = slow->next;
        fast->next = p;
        pre->next = NULL;
        return slow;
    }
};
