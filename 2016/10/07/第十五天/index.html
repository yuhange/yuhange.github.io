设计一个有getMin功能的栈

实现一个特殊的栈，在实现栈的基本功能的基础上，再实现返回栈中最小元素的操作。

要求：

  • pop, push, getMin的时间复杂度都是O(1)
  • 设计的栈类型可以使用现成的栈结构

解析：

  • 使用一个辅助栈来存放当前最小值和历史最小值。
class Solution {
public:
    void push(int value) {
        data.push(value);
        if(m_min.empty() || m_min.top()>value)
            m_min.push(value);
        else 
            m_min.push(m_min.top());
    }
    void pop() {
        data.pop();
        m_min.pop();
    }
    int top() {
        return data.top();
    }
    int min() {
        return m_min.top();
    }
    stack<int> data;
    stack<int> m_min;
};

解法二：

class Solution {
public:
    void push(int value) {
        data.push(value);
        if(m_min.empty() || m_min.top()>value)
            m_min.push(value);
    }
    int pop() {
        int val = data.top();
        data.pop();
        if(!m_min.empty() && m_min.top()==val)
            m_min.pop();
        return val;
    }
    int top() {
        return data.top();
    }
    int min() {
        return m_min.top();
    }
    stack<int> data;
    stack<int> m_min;
};

由两个栈组成的队列

两个栈实现队列的pop, push, front 函数

解析：

  • 两个栈，一个用作压入，一个用作弹出。
  • 当弹出时，如果弹出栈为空，将压入栈的所有元素压入弹出栈中。否则直接输出弹出栈的栈顶。
class Solution{
public:
    void push(int node) {
        s1.push(node);
    }
    int pop() {
        if(s2.empty()){
            while(!s1.empty()){
                s2.push(s1.top());
                s1.pop();
            }
        }
        int res = s2.top();
        s2.pop();
        return res;
    }
    int front(){
        if(s2.empty()){
            while(!s1.empty()){
                s2.push(s1.top());
                s1.pop();
            }
        }
        int res = s2.top();
        return res;
    }
private:
    stack<int> s1;
    stack<int> s2;
};
