打印两个有序链表的公共部分

给定两个有序链表head1和head2，打印两个链表的公共部分。

package chapter_2_listproblem;
public class Problem_01_PrintCommonPart {
    /**打印两个有序链表的公共部分**/
    static void print(Node n1, Node n2){
        if(n1 == null || n2 == null) return;
        Node p1 = n1;
        Node p2 = n2;
        while(p1!=null && p2!=null){
            if(p1.value>p2.value) p2 = p2.next;
            else if(p1.value<p2.value) p1 = p1.next;
            else {
                System.out.println(p1.value);
                p1 = p1.next;
                p2 = p2.next;
            }
        }
    }
    public static void main(String[] arg){
        Node n1 = new Node(1);
        Node n2 = new Node(7);
        Node n3 = new Node(9);
        Node n4 = new Node(14);
        Node n5 = new Node(15);
        Node n6 = new Node(17);
        n1.next = n2;
        n2.next = n3;
        n3.next = n4;
        n4.next = n5;
        n5.next = n6;
        Node n10 = new Node(1);
        Node n20 = new Node(2);
        Node n30 = new Node(9);
        Node n40 = new Node(14);
        Node n50 = new Node(25);
        Node n60 = new Node(36);
        n10.next = n20;
        n20.next = n30;
        n30.next = n40;
        n40.next = n50;
        n50.next = n60;
        print(n1, n10);
    }
}

在单链表和双链表中删除倒数第K个结点

分别实现两个函数，一个可以删除单链表中倒数第K个结点，另一个可以删除双链表中倒数第K的结点。

package chapter_2_listproblem;
public class Problem_02_RemoveLastKthNode {

    static Node deleteKthNodeFromEnd(Node node, int k){
        if(node == null || k<1) return node;
        Node n1 = node;
        Node n2 = node;
        while(n2.next!=null && k>0){
            n2 = n2.next;
            k--;
        }
        while(n2.next!=null){
            n2 = n2.next;
            n1 = n1.next;
        }
        n1.next = n1.next.next;
        return node;
    }
    public static void main(String[] args) {
        // TODO Auto-generated method stub
        Node n1 = new Node(1);
        Node n2 = new Node(7);
        Node n3 = new Node(9);
        Node n4 = new Node(14);
        Node n5 = new Node(15);
        Node n6 = new Node(17);
        n1.next = n2;
        n2.next = n3;
        n3.next = n4;
        n4.next = n5;
        n5.next = n6;
        deleteKthNodeFromEnd(n1, 3);
        n1.print(n1);
    }
}

删除链表的中间结点和a/b处的结点

package chapter_2_listproblem;
public class Problem_03_RemoveNodeByRatio {

    static Node removeMidNode(Node n){
        if(n == null || n.next == null) return n;
        Node fast = n.next.next;//注意这里
        Node slow = n;
        while(fast.next != null && fast.next.next!=null){
            fast = fast.next.next;
            slow = slow.next;
        }
        slow.next = slow.next.next;
        return n;
    }

    public static void main(String[] args){
        Node n1 = new Node(1);
        Node n2 = new Node(7);
        Node n3 = new Node(9);
        Node n4 = new Node(14);
        Node n5 = new Node(15);
        Node n6 = new Node(17);
        Node n7 = new Node(19);
        n1.next = n2;
        n2.next = n3;
        n3.next = n4;
        n4.next = n5;
        n5.next = n6;
        n6.next = n7;
        removeMidNode(n1);
        n1.print(n1);
    }
}

反转单向和双向链表

package chapter_2_listproblem;
public class Problem_04_ReverseList {
    static Node reverse(Node head){
        if(head == null || head.next == null) return head;
        Node pre = null;
        Node next = null;
        while(head!=null){
            next = head.next;
            head.next = pre;//每次都是在下一步时将head指向前一个。
            pre = head;
            head = next;
        }
        return pre;
    }
    public static void main(String[] args) {
        // TODO Auto-generated method stub
        Node n1 = new Node(1);
        Node n2 = new Node(7);
        Node n3 = new Node(9);
        Node n4 = new Node(14);
        Node n5 = new Node(15);
        Node n6 = new Node(17);
        n1.next = n2;
        n2.next = n3;
        n3.next = n4;
        n4.next = n5;
        n5.next = n6;
        Node n = reverse(n1);
        n.print(n);
    }
}
