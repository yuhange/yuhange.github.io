package sort;
/**快速排序算法**/
public class QuickSort {

    static void sort(int[] nums, int begin, int end){
        if(begin<end){
            int index = partition(nums, begin, end);
            sort(nums, 0, index-1);
            sort(nums, index+1, end);
        }
    }
    static int partition(int[] nums, int begin, int end){
        int pivot = nums[end];//用最后一个数字作为基准
        int index = end;
        //未改进前算法：
        /*for(int i = begin; i<=end; i++){
            if(nums[i]>pivot && i<index){
                index = i;
                int tmp = nums[i];
                nums[i] = nums[end];
                nums[end] = tmp;
            }
            else if(nums[i]<pivot && i>index){
                int tmp = nums[i];
                nums[i] = nums[index];
                nums[index] = tmp;
                index = i;
            }
        }*/
        //改进算法：
        for(int i = end-1; i>=begin; i--){
            if(nums[i]>pivot){
                index--;
                if(index!=i){
                    int tmp = nums[index];
                    nums[index] = nums[i];
                    nums[i] = tmp;
                }
            }
        }
        nums[end] = nums[index];
        nums[index] = pivot;
        return index;
    }

}

package chapter_2_listproblem;
/**反转部分单链表**/
public class Problem_05_ReversePartList {
    /**思路：获得反转序列的前一个和后一个结点，反转要反转的序列，连接。**/
    static Node reversePart(Node head, int from, int to){
        if(head == null || head.next==null || to<=from) return head;
        int i = 1;
        Node pre = head;//反转序列第一个结点的前一个结点
        Node next = head;//反转序列的最后一个结点的后一个结点
        for(; i<from && next.next!=null; i++){
            pre = next;
            next = next.next;
        }
        for(i = from; i<=to && next!=null; i++){
            next = next.next;
        }
        //System.out.println(pre.value);
        Node p = null;//这是反转后的第一个结点，用它和pre连接
        Node n = null;
        Node cur = pre.next; //这是反转后的最后一个结点，用它和next连接。
        Node tmp = pre.next;
        for(i = from; i<=to && tmp!=null; i++){
            n = tmp.next;
            //if(n!=null) 
                tmp.next = p;
            p = tmp;
            tmp = n;
            //System.out.println(p.value);
        }
        cur.next = next;
        pre.next = p;
        //System.out.println(p.value);
        return head;
    }
    public static void main(String[] args) {
        // TODO Auto-generated method stub
        Node n1 = new Node(1);
        Node n2 = new Node(7);
        Node n3 = new Node(9);
        Node n4 = new Node(14);
        Node n5 = new Node(15);
        Node n6 = new Node(17);
        n1.next = n2;
        n2.next = n3;
        n3.next = n4;

    }
}

环形单链表的约瑟夫问题

O(N*M)的算法

package chapter_2_listproblem;
public class Problem_06_JosephusProblem {
    static Node josephusKill(Node head, int m){
        if(head == null || head.next == head || m<1) return head;
        Node last = head;
        while(last.next!=head) last = last.next; //将last置于head的前一个。
        int count = 0;
        while(head!=last){
            if(++count == m){
                count = 0;
                last.next = head.next;
            }else{
                last = last.next;
            }
            head = last.next;
        }
        return head;
    }
    public static void main(String[] args) {
        // TODO Auto-generated method stub
        Node n1 = new Node(1);
        Node n2 = new Node(2);
        Node n3 = new Node(3);
        Node n4 = new Node(4);
        Node n5 = new Node(5);
        Node n6 = new Node(6);
        Node n7 = new Node(7);
        Node n8 = new Node(8);
        Node n9 = new Node(9);
        Node n10 = new Node(10);
        n1.next = n2;
        n2.next = n3;
        n3.next = n4;
        n4.next = n5;
        n5.next = n6;
        n6.next = n7;
        n7.next = n8;
        n8.next = n9;
        n9.next = n10;
        n10.next = n1;
        Node n = josephusKill(n1, 3);
        System.out.println(n.value);
    }
}

判断一个单链表是否是回文结构

方法一

package chapter_2_listproblem;
import java.util.Stack;
public class Problem_07_IsPalindromeList {
    static boolean isPalindrome(Node head){
        if(head == null || head.next == null) return true;
        Stack<Node> s = new Stack<Node>();
        Node p = head;
        while(p!=null){
            s.push(p);
            p = p.next;
        }
        p = head;
        while(!s.isEmpty() && p!=null){
            int v = s.pop().value;
            if(v!= p.value) {
                return false;
            }
            p = p.next;
        }
        return true;
    }
    public static void main(String[] args) {
        // TODO Auto-generated method stub
        Node n1 = new Node(1);
        Node n2 = new Node(2);
        Node n3 = new Node(3);
        Node n4 = new Node(4);
        Node n5 = new Node(5);
        Node n6 = new Node(4);
        Node n7 = new Node(3);
        Node n8 = new Node(2);
        Node n9 = new Node(1);
        //Node n10 = new Node(10);
        n1.next = n2;
        n2.next = n3;
        n3.next = n4;
        n4.next = n5;
        n5.next = n6;
        n6.next = n7;
        n7.next = n8;
        n8.next = n9;
        //n9.next = n10;
        System.out.println(isPalindrome(n1));
    }
}
