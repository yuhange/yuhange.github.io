实现两个相加链表

package chapter_2_listproblem;
public class Problem_10_AddTwoLinkedLists {
    static Node add(Node h1, Node h2){
        Node head = null;
        Node p = null;
        Node head1 = reverse(h1);
        Node head2 = reverse(h2);
        int ca = 0;
        while(head1!=null || head2!=null){
            int a = head1==null?0:head1.value;
            int b = head2==null?0:head2.value;
            int c = a+b+ca;
            ca = c/10;
            p = head;
            head = new Node(c%10);
            head.next = p;
            head1 = head1==null?null:head1.next;
            head2 = head2==null?null:head2.next;
        }
        if(ca != 0){
            p = head;
            head = new Node(1);
            head.next = p;
        }
        h1 = reverse(h1);
        h2 = reverse(h2);
        return head;
    }
    static Node reverse(Node h){
        Node pre = null;
        Node next = null;
        while(h!=null){
            next = h.next;
            h.next = pre;
            pre = h;
            h = next;
        }
        return pre;
    }
    public static void main(String[] args) {
        // TODO Auto-generated method stub
        Node n1 = new Node(9);
        Node n2 = new Node(9);
        Node n3 = new Node(9);
        Node n4 = new Node(0);
        Node n5 = new Node(3);
        n1.next = n2;
        n2.next = n3;
        n3.next = null;
        //n3.next = n4;
        n4.next = n5;
        n5.next = null;
        Node head = add(n1, n4);
        head.print(head);
    }
}

将单链表的每K个结点之间逆序

package chapter_2_listproblem;
public class Problem_12_ConvertEveryKNodesInList {
    static Node reverse(Node head, int k){
        if(k<2) return head;
        Node cur = head;
        Node pre = null;
        Node next = null;
        Node start = head;
        int count = 1;
        while(cur != null){
            next = cur.next;
            if(count == k){
                count = 0;
                start = pre == null?head:pre.next;
                head = pre == null?cur:head;
                reverse(pre, start, cur, next);
                pre = start;
                count = 0;
            }
            count++;
            cur = next;
        }
        return head;
    }
    static void reverse(Node left, Node start, Node end, Node right){
        Node pre = start;
        Node cur = start.next;
        Node next = null;
        while(cur!=right){
            next = cur.next;
            cur.next = pre;
            pre = cur;
            cur = next;
        }
        if(left!=null) left.next = end;
        start.next = right;
    }
    public static void main(String[] args) {
        // TODO Auto-generated method stub
        Node n1 = new Node(1);
        Node n2 = new Node(2);
        Node n3 = new Node(3);

删除无序单链表中值重复出现的结点

package chapter_2_listproblem;
import java.util.*;
public class Problem_13_RemoveRepetition {
    static Node delete(Node head){
        if(head == null) return head;
        HashSet<Integer> set = new HashSet<Integer>();
        Node pre = head;
        Node cur = head.next;
        set.add(cur.value);
        while(cur!=null){
            //System.out.println(cur.value);
            if(!set.contains(cur.value)) {
                set.add(cur.value);
                pre = cur;
            }
            else{
                pre.next = cur.next;
            }
            cur = cur.next;
        }
        return head;
    }
    public static void main(String[] args) {
        // TODO Auto-generated method stub
        Node n1 = new Node(9);
        Node n2 = new Node(9);
        Node n3 = new Node(9);
        Node n4 = new Node(0);
        Node n5 = new Node(3);
        n1.next = n2;
        n2.next = n3;
        n3.next = n4;
        n4.next = n5;
        Node n = delete(n1);
        n.print(n);
    }
}

删除单链表中指定值的结点

package chapter_2_listproblem;
public class Problem_14_RemoveGivenValue {
    static Node delete(Node head, int k){
        while(head!=null){
            if(head.value != k) break;
            head = head.next;
        }
        Node cur = head;
        Node pre = head;
        while(cur!=null){
            if(cur.value == k) {
                pre.next = cur.next;
            }else{
                pre = cur;
            }
            cur = cur.next;
        }
        return head;
    }
    public static void main(String[] args) {
        // TODO Auto-generated method stub
        Node n1 = new Node(9);
        Node n2 = new Node(9);
        Node n3 = new Node(9);
        Node n4 = new Node(0);
        Node n5 = new Node(3);
        n1.next = n2;
        n2.next = n3;
        n3.next = n4;
        n4.next = n5;
        Node n = delete(n1, 0);
        n.print(n);
    }
}
