inner_product

计算[first1, last1)和[first2, first2+(last1-first1))的一般内积。要提供一个init值，以备当[first, last)为空时也有一个明确的返回值。

template <class InputIterator1, class InputIterator2, class T>
T inner_product(InuptIterator1 first1, InputIterator1 last1, InputIterator2 first2, T init){
    for(; first1 != last1; ++first1, ++first2)[
        init += (*first1 * *first2);
    }
    return init;
}

equal

判断两个序列在[first, last)之间的元素是否完全相同。如果序列2的长度长于了序列一，多出的部分不予考虑。

template <class InputIterator1, class InputIterator2>
inline bool equal(InputIterator1 first1, InputIterator1 last1, InputIterator2 first2){
    for(; first1 != last1 ; first1++, first2++){
        if(*first1 != *first2) return false;
    }
    return true;
}

void fill(ForwardIterator first, ForwardIterator last, const T& value){
    for(; first!=last; first++)
        *first = value;
}

lexicographical_compare

以“字典排序方式”对两个序列[first1, last1)和[first2, last2)进行比较。

比较操作符针对两序列中对应位置的元素进行，直到：

  • 某一组对应元素彼此不等
  • 同时到达last1和last2
  • 到达last1或last2

返回值：

  • 如果第一个序列较小，返回true
  • 如果到达last1而没有到达last2， 返回false
  • 如果到达last2而没有到达last1， 返回true
  • 同时到达last1和last2，返回false。
    综上， 第一个序列只要不大于第二个序列则返回false。
template <class InputIterator1, class InputIterator2>
bool lexicographical_compare(InputIterator1 first1, InputIterator1 last1, InputIterator2 first2, InputIterator2 last2){
    for(; first1!=last1 && first2!=last2; ++first1, ++first2){
        if(*first1<*first2) return true;//出现小于字符的就可以判定小于了
        if(*first1>*first2) return false;
    }   
    return first1 == last1 && first2 != last2;
}

66. Plus One

Given a non-negative number represented as an array of digits, plus one to the number.

The digits are stored such that the most significant digit is at the head of the list.

简单题，但是没有一次做对。

解答：

class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        vector<int> res;
        int n = digits.size();
        if(n==0) return res;
        if(n == 1 && digits[0]<9) {res.push_back(digits[0]+1); return res;}
        int ca = 0;
        for(int i = n-1; i>=0; i--){
            int a = digits[i];
            int b = i==n-1?1:0;
            int c = a+b+ca;
            res.push_back(c%10);
            ca = c/10;
            //cout<<"c "<<c<<" ca: "<<ca<<endl;
        }
        if(ca == 1){
            res.push_back(1);
        }
        reverse(res.begin(),res.end());
        return res;
    }
};

88. Merge Sorted Array

Given two sorted integer arrays nums1 and nums2, merge nums2 into nums1 as one sorted array.

Note:

You may assume that nums1 has enough space (size that is greater or equal to m + n) to hold additional elements from nums2. The number of elements initialized in nums1 and nums2 are m and n respectively.

十分简单的入门题竟然搞了好久最后也没做出来。

解答：

class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        int i = m-1;
        int j = n-1;
        int k = m+n-1;
        while(i>=0 && j>=0){
            if(nums1[i]>nums2[j]){
                nums1[k--] = nums1[i--];
            }else
                nums1[k--] = nums2[j--];
        }
        while(j>=0){
            nums1[k--] = nums2[j--];
        }
    }
};

242. Valid Anagram

Given two strings s and t, write a function to determine if t is an anagram of s.

For example,
s = “anagram”, t = “nagaram”, return true.
s = “rat”, t = “car”, return false.

Note:

You may assume the string contains only lowercase alphabets.

Follow up:

What if the inputs contain unicode characters? How would you adapt your solution to such case?

因为一个变量写错WA了一次！

解答：

class Solution {
public:
    bool isAnagram(string s, string t) {
        int n = s.length();
        int m = t.length();
        if(n!=m) return false;
        int hashmap[256] = {0};
        for(int i = 0; i<n; i++)
            hashmap[s[i]] ++;
        for(int i = 0; i<m; i++)
            hashmap[t[i]] --;
        for(int i = 0; i<256; i++)
            if(hashmap[i]!=0) return false;
        return true;
    }
};

237. Delete Node in a Linked List

Write a function to delete a node (except the tail) in a singly linked list, given only access to that node.

Supposed the linked list is 1 -> 2 -> 3 -> 4 and you are given the third node with value 3, the linked list should become 1 -> 2 -> 4 after calling your function.

技巧题，记住方法即可。

解答：

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    void deleteNode(ListNode* node) {
        if(node == NULL) return;
        node->val = node->next->val;
        node->next = node->next->next;
    }
};

33. Search in Rotated Sorted Array

Suppose a sorted array is rotated at some pivot unknown to you beforehand.

(i.e., 0 1 2 4 5 6 7 might become 4 5 6 7 0 1 2).

You are given a target value to search. If found in the array return its index, otherwise return -1.

You may assume no duplicate exists in the array.

当年觉得很难的题竟然可以自己写出来啦！但是还是WA了一次，原因是考虑不全面。

现在的代码有优化余地。

解答：

class Solution {
public:
    int search(vector<int>& nums, int target) {
        int i = 0;
        int n = nums.size();
        if(n == 0) return -1;
        for(; i<n-1; i++) if(nums[i]>nums[i+1]) break;
        cout<<i<<endl;
        if(nums[i] == target) return i;
        if(nums[i] < target) return -1;
        //if(nums[n-1]<target) return search(nums, target, 0, i-1);
        //if(nums[0]>target) return search(nums, target, i+1, n-1);
        int left = search(nums, target, 0, i-1);
        int right = search(nums, target, i+1, n-1);
        if(left == -1 && right == -1) return -1;
        if(left != -1) return left;
        if(right != -1) return right;
        return -1;
    }
    int search(vector<int>& nums, int target, int l, int r){
        int mid = 0;
        cout<<"l: "<<l<<" r: "<<r<<endl;

            if(nums[mid] == target) return mid;
            if(nums[mid]>target) r = mid-1;
            else l = mid+1;
        }
        return -1;
    }
};

34. Search for a Range

Given a sorted array of integers, find the starting and ending position of a given target value.

Your algorithm’s runtime complexity must be in the order of O(log n).

If the target is not found in the array, return [-1, -1].

For example,

Given [5, 7, 7, 8, 8, 10] and target value 8,
return [3, 4].

同样，当年觉得难得不可能做出来的题目也是自己做的！一次AC！

解答：

class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        vector<int> res = {-1, -1};
        int n = nums.size();
        if(n == 0) return res;
        //先找最右，再找最左（出现的）
        int ll = 0;
        int lr = n-1;
        int mid = 0;
        while(ll<=lr){
            mid = ll + (lr-ll)/2;
            if(nums[mid] >= target) lr = mid-1;
            else ll = mid+1;
        }
        //cout<<ll<<" "<<lr<<endl;
        int rl = 0;
        int rr = n-1;
        while(rl<=rr){
            mid = rl + (rr-rl)/2;
            if(nums[mid] <= target) rl = mid+1;
            else rr = mid-1;
        }
        if(ll<=rr) {
            res[0] = ll;
            res[1] = rr;
        }
        //cout<<rl<<" "<<rr<<endl;
        return res;
    }
};

35. Search Insert Position

Given a sorted array and a target value, return the index if the target is found. If not, return the index where it would be if it were inserted in order.

You may assume no duplicates in the array.

Here are few examples.

[1,3,5,6], 5 → 2
[1,3,5,6], 2 → 1
[1,3,5,6], 7 → 4
[1,3,5,6], 0 → 0

这种难度的Medium对于我来说已经和Easy一样了。比好多Easy还要简单。

解答：

class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int n = nums.size();
        //if(n == 0) return -1;
        int l = 0;
        int r = n-1;
        int mid = 0;
        while(l<=r){
            mid = l + (r-l)/2;
            if(nums[mid] >= target) r = mid-1;
            else l = mid+1;
        }
        //cout<<(l+1)<<endl;
        return l;
    }
};
