42. Trapping Rain Water

Given n non-negative integers representing an elevation map where the width of each bar is 1, compute how much water it is able to trap after raining.

For example,

Given [0,1,0,2,1,0,1,3,2,1,2,1], return 6.

解答：

class Solution {
public:
    int trap(vector<int>& a) {
        int n = a.size();
        if(n<=2) return 0;
        int i = 0;
        int j = n-1;
        int total = 0;
        while(i<j){
            int minL = min(a[i], a[j]);
            if(a[i] == minL)
                while(++i < j && a[i]<minL) total += (minL-a[i]);
            else while(i < --j && a[j]<minL) total += (minL-a[j]);
        }
        return total;
    }
};

48. Rotate Image

You are given an n x n 2D matrix representing an image.

Rotate the image by 90 degrees (clockwise).

Follow up:
Could you do this in-place?

解答一：我自己的解法

class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        vector<vector<int>> res;
        int n = matrix.size();
        for(int i = 0; i<n; i++){
            vector<int> v;
            for(int j = 0; j<n; j++){
                cout<<matrix[n-j-1][i]<<" ";
                v.push_back(matrix[n-j-1][i]);
            }
            cout<<endl;
            res.push_back(v);
        }
        matrix = res;
    }
};

解答二：就地

class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        int n = matrix.size();
        //按对称轴交换
        for(int i = 0; i<n; i++)
            for(int j = 0; j<i; j++)
                swap(matrix[i][j], matrix[j][i]);
        //按列交换
        for(int i = 0, j = n-1; i<j; i++, j--)
            for(int k = 0; k<n; k++){
                swap(matrix[k][j], matrix[k][i]);
            }
    }
};
