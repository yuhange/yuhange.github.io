383. Ransom Note

Given an arbitrary ransom note string and another string containing letters from all the magazines, write a function that will return true if the ransom note can be constructed from the magazines ; otherwise, it will return false. Each letter in the magazine string can only be used once in your ransom note.
Note:
You may assume that both strings contain only lowercase letters. 

canConstruct("a", "b") -> false
canConstruct("aa", "ab") -> false
canConstruct("aa", "aab") -> true

本题和判断两个字符串是否互为变位词的解法是相同的。都是判断一个字符串和另一个字符串中的字符数是否相同，或是一个字符串的字符数比另一个字符串中对应的字符数少。

class Solution {
public:
    bool canConstruct(string ransomNote, string magazine) {
        int t[26] = {0};
        for(int i = 0; i<magazine.length(); i++){
            int c_i = magazine[i]-'a';
            t[c_i]++;
        }
        for(int i = 0; i<ransomNote.length(); i++){
            int c_i = ransomNote[i]-'a';
            if(--t[c_i]<0) return false;
        }
        return true;
    }
};

345. Reverse Vowels of a String

Write a function that takes a string as input and reverse only the vowels of a string.

Example 1:
Given s = “hello”, return “holle”.

Example 2:
Given s = “leetcode”, return “leotcede”.

Note:
The vowels does not include the letter “y”.

双指针问题

class Solution {
public:
    string reverseVowels(string s) {
        int n = s.length();
        if(n == 0 || n==1) return s;
        int i = 0; 
        int j = n-1;
        while(i<j){
            if(!isVowel(s[i])) i++;
            if(!isVowel(s[j])) j--;
            if(isVowel(s[i]) && isVowel(s[j])){
                char t = s[i];
                s[i] = s[j];
                s[j] = t;
                i++;
                j--;
            }
        }
        return s;
    }
    bool isVowel(char c){
        if(c=='a' || c=='e' || c=='o' || c=='i'|| c=='u'|| c=='A' || c=='E' || c=='O' || c=='I'|| c=='U'){
            return true;
        }
        return false;
    }
};

344. Reverse String

Write a function that takes a string as input and returns the string reversed.

Example:
Given s = “hello”, return “olleh”.

双指针问题

class Solution {
public:
    string reverseString(string s) {
        int n = s.length();
        if(n == 0 || n==1) return s;
        int i = 0; 
        int j = n-1;
        while(i<j){
            char t = s[i];
            s[i] = s[j];
            s[j] = t;
            i++;
            j--;
        }
        return s;
    }
};

125. Valid Palindrome

Given a string, determine if it is a palindrome, considering only alphanumeric characters and ignoring cases.

For example,
“A man, a plan, a canal: Panama” is a palindrome.
“race a car” is not a palindrome.

Note:
Have you consider that the string might be empty? This is a good question to ask during an interview.

For the purpose of this problem, we define empty string as valid palindrome. 

class Solution {
public:
    bool isPalindrome(string s) {
        int n = s.length();
        if(n == 0 || n==1) return true;
        int i =0, j=n-1;
        while(i<j){
            if(isUpperCase(s[i])) s[i] = s[i]+32;
            if(isUpperCase(s[j])) s[j] = s[j]+32;
            if(!isLegal(s[i])) i++;
            if(!isLegal(s[j])) j--;
            if(isLegal(s[i])&&isLegal(s[j])){
                if(s[i]!=s[j]) return false;
                i++;
                j--;
            }
        }
        return true;
    }
    bool isUpperCase(char c){
        if(c<='Z'&&c>='A') return true;
        return false;
    }
    bool isLegal(char c){
        if(c>='a'&&c<='z' || c>='0'&&c<='9') return true;
        return false;
    }
};

58. Length of Last Word

Given a string s consists of upper/lower-case alphabets and empty space characters ‘ ‘, return the length of last word in the string.

If the last word does not exist, return 0.

Note: A word is defined as a character sequence consists of non-space characters only.

For example,
Given s = “Hello World”,
return 5. 

class Solution {
public:
    int lengthOfLastWord(string s) {
        int n = s.length();
        if(n == 0) return 0;
        if(n == 1 && s[n-1]==' ') return 0;
        if(n == 1 && s[n-1]!=' ') return 1;
        int i = n-1;
        int count = 0;
        while(i>=0&&s[i]==' ')
            i--;
        while(i>=0&&s[i]!=' '){
            i--;
            count++;
        }

        return count;
    }
};

28. Implement strStr()

Implement strStr().

Returns the index of the first occurrence of needle in haystack, or -1 if needle is not part of haystack. 

class Solution {
public:
    int strStr(string haystack, string needle) {
        int m = haystack.size();
        int n = needle.size();
        for(int i = 0; i <= m-n; i ++)
        {
            int j;
            for(j = 0; j < n; j ++)
            {
                if(haystack[i+j] != needle[j])
                    break;
            }
            if(j == n)
                return i;
        }
        return -1;
    }
};

20. Valid Parentheses

Given a string containing just the characters ‘(‘, ‘)’, ‘{‘, ‘}’, ‘[‘ and ‘]’, determine if the input string is valid.

The brackets must close in the correct order, “()” and “()[]{}” are all valid but “(]” and “([)]” are not.

class Solution {
public:
    bool isValid(string s) {
        int n = s.length();
        if(n == 0 || n%2 ==1) return false;
        stack<char> stack;
        for(int i = 0; i<n ; i++){
            if(shouldPush(s[i])) stack.push(s[i]);
            if(shouldPop(s[i])){
                if(stack.empty()) return false;
                if(shouldPop(stack.top(), s[i])) stack.pop();
            }
        }
        if(stack.empty()) return true;
        return false;
    }
    bool shouldPush(char c){
        if(c=='{'||c=='('||c=='[') return true;
        return false;
    }
    bool shouldPop(char out){
        if(out=='}' ||out==']' ||out==')') return true;
        return false;
    }
    bool shouldPop(char in, char out){
        if(in=='{'&&out=='}' ||in=='['&&out==']' ||in=='('&&out==')') return true;
        return false;
    }
};
