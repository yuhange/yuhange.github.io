两个链表的第一个公共结点

/*
struct ListNode {
    int val;
    struct ListNode *next;
    ListNode(int x) :
            val(x), next(NULL) {
    }
};*/
class Solution {
public:
    ListNode* FindFirstCommonNode( ListNode *pHead1, ListNode *pHead2) {
        int len1 = getLength(pHead1);
        int len2 = getLength(pHead2);
        ListNode* pLong = len1>len2?pHead1:pHead2;
        ListNode* pShort = len1>len2?pHead2:pHead1;
        int diff = len1>len2?(len1-len2):(len2-len1);
        while(diff>0) {pLong = pLong->next;diff--;}
        bool ok = false;
        while(pLong!=NULL && pShort!=NULL){
            if(pLong == pShort) {ok = true; break;}
            pLong = pLong->next;
            pShort = pShort->next;
        }
        if(ok) return pLong;
        return NULL;
    }
    int getLength(ListNode* pHead){
        int count = 0;
        ListNode* p = pHead;
        while(p!=NULL){
            p = p->next;
            count++;
        }
        return count;
    }
};

链表中环的入口结点 （剑指OFFER+leetcode）

一个链表中包含环，请找出该链表的环的入口结点。

/*
struct ListNode {
    int val;
    struct ListNode *next;
    ListNode(int x) :
        val(x), next(NULL) {
    }
};
*/
class Solution {
public:
    ListNode* EntryNodeOfLoop(ListNode* pHead)
    {
        //链表中包含环所以不用担心头结点为空的情况了。
        ListNode* fast = pHead;
        ListNode* slow = pHead;
        while(fast!=NULL && fast->next!=NULL){
            fast = fast->next->next;
            slow = slow->next;
            if(fast == slow){
                break;
            }
        }
        if(fast!=NULL && fast->next!=NULL){
            slow = pHead;
            while(slow!=fast){
                slow = slow->next;
                fast = fast->next;
            }
            return slow;
        }
        return NULL;
    }
};
