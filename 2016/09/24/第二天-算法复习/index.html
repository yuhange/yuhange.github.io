int Joseph(int n, int m){
    int J[100] = {-1};
    for(int i = 1; i<=n;i++){
        J[i] = i;
    }
    int in = n;
    int count = 0;
    while(in>0){
        for(int i = 1; i<=n; i++){
            if(J[i]!=-1){
                count++;
                if(count == m){
                    J[i] = -1;
                    count = 0;
                }
            }

        }
    }
    for(int i = 1 ; i<= n; i++){
        if(J[i] != -1) return i;
    }
    return 0;
}

剑指OFFER 面试题1 二维数组中的查找

题目描述：在一个二维数组中，每一行都按照从左到右递增的顺序排序，每一列都按照从上到下递增的顺序排序。请完成一个函数，输入这样的一个二维数组和一个整数，判断数组中是否含有该整数。

class Solution {
public:
    bool Find(vector<vector<int> > array,int target) {
        if(array.size() == 0) return false;
        int n = array[0].size();
        int m = array.size();
        int i = 0;
        int j = n-1;
        int p = 0;
        bool result = false;
        while( i<= n-1 && j>=0 ){
            p = array[i][j];
            if(target == p) {result = true; break;}
            else if(target<p) j--;
            else i++;
        }
        return result;
    }
};

这段代码的while循环的判断条件还是会搞不清楚，但是基本的思路是有的了。

剑指OFFER 面试题2：替换空格

题目描述：请实现一个函数，将一个字符串中的空格替换成“%20”。例如，当字符串为We Are Happy.则经过替换之后的字符串为We%20Are%20Happy。

//length为牛客系统规定字符串输出的最大长度，固定为一个常数
class Solution {
public:
    void replaceSpace(char *str,int length) {
        if(str == NULL || length<0) return;
        int len = 0;
        int spaceNum = 0;
        int i = 0;
        while(str[i]!='\0'){
            len++;
            if(str[i]==' ') spaceNum++;
            i++;
        }
        int len2 = len + spaceNum * 2;
        if(len2 > length) return;
        while(len>=0 && len2>len){
            if(str[len] == ' '){
                str[len2--] = '0';
                str[len2--] = '2';
                str[len2--] = '%';
            }else{
                str[len2--] = str[len];
            }
            len--;
        }
    }
};

剑指OFFER 面试题3：从尾到头打印链表

题目描述：输入一个链表，从尾到头打印链表每个节点的值。

非递归算法：
/**
*  struct ListNode {
*        int val;
*        struct ListNode *next;
*        ListNode(int x) :
*              val(x), next(NULL) {
*        }
*  };
*/
class Solution {
public:
    vector<int> printListFromTailToHead(struct ListNode* head) {
        vector<int> res;
        if(head == NULL) return res;
        stack<int> s;
        ListNode* p = head;
        while(p != NULL){
            s.push(p->val);
            p = p->next;
        }
        while(!s.empty()){
            res.push_back(s.top());
            s.pop();
        }
        return res;
    }
};
递归算法：
class Solution {
private:
    vector<int> res;
public:
    vector<int> printListFromTailToHead(struct ListNode* head) {
        if(head==NULL) return res;
        printListFromTailToHead(head->next);
        res.push_back(head->val);
        return res;
    }
};

剑指OFFER 面试题4：重建二叉树

题目描述：输入某二叉树的前序遍历和中序遍历的结果，请重建出该二叉树。假设输入的前序遍历和中序遍历的结果中都不含重复的数字。例如输入前序遍历序列{1,2,4,7,3,5,6,8}和中序遍历序列{4,7,2,1,5,3,8,6}，则重建二叉树并返回。

/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    struct TreeNode* reConstructBinaryTree(vector<int> pre,vector<int> in) {
        if(pre.size() == 0 || in.size() == 0) return NULL;
        TreeNode* root = reConstructBinaryTree(pre, in, 0, pre.size()-1, 0, in.size()-1);
        return root;
    }
    TreeNode* reConstructBinaryTree(vector<int> pre, vector<int> in, int preS, 
                                    int preE, int inS, int inE){
        if(preS>preE || inS>inE) return NULL;
        TreeNode* root = new TreeNode(pre[preS]);
        root->left = root->right = NULL;
        for(int i = inS; i<=inE; i++){
            if(in[i] == pre[preS]){
                root->left = reConstructBinaryTree(pre, in, preS+1, preS+i-inS, inS, i-1);
                root->right = reConstructBinaryTree(pre, in, preS+i-inS+1, preE, i+1, inE);
                break;
            }
        }
        return root;
    }

};

剑指OFFER 面试题5：用两个栈实现队列

题目描述：用两个栈来实现一个队列，完成队列的Push和Pop操作。 队列中的元素为int类型。

class Solution
{
public:
    void push(int node) {
        s1.push(node);
    }

    int pop() {
        if(s2.empty()){
            while(!s1.empty()){
                s2.push(s1.top());
                s1.pop();
            }
        }
        int res = s2.top();
        s2.pop();
        return res;
    }

private:
    stack<int> s1;
    stack<int> s2;
};

剑指OFFER 面试题6：旋转数组的最小值（二分法）

题目描述：把一个数组最开始的若干个元素搬到数组的末尾，我们称之为数组的旋转。

class Solution {
public:
    int minNumberInRotateArray(vector<int> array) {
        int n = array.size();
        if(n == 0) return 0;
        int i = 0, j = n-1;
        int mid = 0;
        while(array[i]>=array[j]){
            if(j-i == 1){
                mid = j;
                break;
            }
            mid = (i+j)/2;
            if(array[mid] == array[i] && array[mid]==array[j]) return minInOrder(array);
            if(array[mid]>=array[i]) i = mid;
            if(array[mid]<=array[j]) j = mid;
        }
        return array[mid];
    }
    int minInOrder(vector<int> array){
        int min = array[0];
        for(int i = 0; i<array.size(); i++){
            min = min<array[i]?min:array[i];
        }
        return min;
    }
};
