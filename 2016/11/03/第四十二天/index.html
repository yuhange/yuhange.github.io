86. Partition List

Given a linked list and a value x, partition it such that all nodes less than x come before nodes greater than or equal to x.

You should preserve the original relative order of the nodes in each of the two partitions.

For example,
Given 1->4->3->2->5->2 and x = 3,

return 1->2->2->4->3->5.

解答：

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* partition(ListNode* head, int x) {
        //重新起两个链表，小于x的挂在p后面，大于x的挂在q后面，最后合并两个链表。
        ListNode* p = new ListNode(0);
        ListNode* pHead = p;
        ListNode* q = new ListNode(0);
        ListNode* qHead = q;
        ListNode* cur = head;
        while( head ) {
            if( head->val < x ) {
                p->next = head;
                p = p->next;
            }else {
                q->next = head;
                q = q->next;
            }
            head = head->next;
        }
        p->next = qHead->next;
        q->next = NULL;
        return pHead->next;
    }
};

83. Remove Duplicates from Sorted List

Given a sorted linked list, delete all duplicates such that each element appear only once.

For example,

Given 1->1->2, return 1->2.
Given 1->1->2->3->3, return 1->2->3.

解答：

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if( !head || !head->next) return head;
        ListNode* p = head;
        ListNode* q = head->next;
        while( q ) {
            if( p->val == q->val ) {
                p->next = q->next;
            }else p = q;
            q = q->next;
        }
        return head;
    }
};

92. Reverse Linked List II

Reverse a linked list from position m to n. Do it in-place and in one-pass.

For example:
Given 1->2->3->4->5->NULL, m = 2 and n = 4,

return 1->4->3->2->5->NULL.

Note:
Given m, n satisfy the following condition:
1 ≤ m ≤ n ≤ length of list.

解答：

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* reverseBetween(ListNode* head, int m, int n) {
        if(m>=n) return head;
        if(!head || !head->next) return head;
        ListNode* cur = head;
        ListNode* next = NULL;
        ListNode* pre = NULL;
        ListNode* preStart = NULL;
        ListNode* start = NULL;
        int interval = n-m+1;
        while( --m ) {//m为1时，不会执行，cur仍为head. preStart仍为NULL.
            preStart = cur;
            cur = cur->next;
        }
        start = cur;
        while( cur && interval-- ) {
            next = cur->next;
            cur->next = pre;
            pre = cur;
            cur = next;
        }
        start->next = next;
        if(preStart==NULL) return pre;
        else preStart->next = pre;
        return head;
    }
};

77. Combinations

Given two integers n and k, return all possible combinations of k numbers out of 1 … n.

For example,
If n = 4 and k = 2, a solution is:

[
  [2,4],
  [3,4],
  [2,3],
  [1,2],
  [1,3],
  [1,4],
]

解答：

class Solution {
public:
    vector<vector<int>> res;
    vector<vector<int>> combine(int n, int k) {
        vector<int> v;
        helper(1, n, k, v);
        return res;
    }
    void helper(int idx, int n, int k, vector<int>& v) {
        if(k == 0) {
            res.push_back(v);
            return;
        }
        for(int i = idx; i<=n; i++){
            v.push_back(i);
            helper(i+1, n, k-1, v);
            v.pop_back();
        }
    }
};

78. Subsets

Given a set of distinct integers, nums, return all possible subsets.

Note: The solution set must not contain duplicate subsets.

For example,
If nums = [1,2,3], a solution is:

[
  [3],
  [1],
  [2],
  [1,2,3],
  [1,3],
  [2,3],
  [1,2],
  []
]

解答：

class Solution {
public:
    vector<vector<int>> res;
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<int> v;
        for(int i = 0; i<=nums.size(); i++) {
            helper(0, nums, 0, i, v);
        }
        return res;
    }
    void helper(int idx, vector<int>& nums, int k, int curSize, vector<int>& v) {
        if(k == curSize) {
            res.push_back(v);
            return;
        }
        for(int i = idx; i<nums.size(); i++){
            v.push_back(nums[i]);
            helper(i+1, nums, k+1, curSize, v);
            v.pop_back();
        }
    }
};
