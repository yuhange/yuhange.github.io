7 Reverse an int

class Solution {
public:
int reverse(int x) {
    long long ans = 0;
    const int maxint = 0x7fffffff;
    const int minint = 0x80000000;
    while( x!=0 ){
        ans = ans*10 + x%10;
        x/=10;
    }
    if(ans<minint || ans>maxint){
        return 0;
    }
    return ans;
}
};

231 Power of Two

Trick

n&(n-1) 去掉n最左边的1.

class Solution {
public:
    bool isPowerOfTwo(int n) {
        if(n<=0) return false;
        return ((n&(n-1))==0);
    }
};

342 Power of Four

Trick

n&0x55555555: 保证1在奇数位。

class Solution {
public:
bool isPowerOfFour(int n) {
        if(n<=0) return false;
        return ((n&(n-1))==0 && (n&0x55555555));
    }
};

326 Power of Three

此方法只对素数有效

class Solution {
public:
bool isPowerOfThree(int n) {
    int maxint = 0x7fffffff;
    if(n <= 0) return false;
    int k = log(maxint)/log(3);
    int big3 = pow(3, k);
    return (big3%n == 0);
}
};

191 Number of 1 Bits

class Solution {
public:
int hammingWeight(uint32_t n) {
    int ans = 0;
    while(n!=0){
        n = n&(n-1);//最高位变0
        ans++;
    }
    return ans;
}
};

172 Factorial Trailing Zeros

class Solution {
public:
int trailingZeroes(int n) {
    int sum = 0;
    while(n>0){
        sum+=n/5;
        n/=5;
    }
    return sum;
}
};

258 Add Digits

class Solution {
public:
int addDigits(int num) {
    if(num == 0) return 0;
    return ((num-1)%9 + 1);
}
};
